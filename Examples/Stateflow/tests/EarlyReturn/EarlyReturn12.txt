%type: module

module Chart():
procedure exec_Chart begin
  Chart_ret := 0;
  if (Chart_st == "Chart_A") {
    Chart_A_done := 0;
    if (ChartEL != [] & top(ChartEL) == "E") {
      if (Chart_st == "Chart_A") {
        Chart_st := "Chart_B";
        s := "enB";
        log("enB" + "\n");
        Chart_A_done := 1;
      }
    }
    Chart_ret := Chart_A_done;
  } else if (Chart_st == "Chart_B") {
    Chart_B_done := 0;
    if (Chart_st == "Chart_B") {
      Chart_st := "Chart_C";
      s := "enC";
      log("enC" + "\n");
      Chart_B_done := 1;
    }
    Chart_ret := Chart_B_done;
  } else if (Chart_st == "Chart_C") {
    Chart_ret := 0;
  }
end

begin
  ChartEL := [];
  Chart_st := "Chart_A";
  ChartEL := push(ChartEL,"E");
  @exec_Chart;
  ChartEL := pop(ChartEL);
  if (Chart_st == "Chart_A") {
    s := "enA_done";
    log("enA_done" + "\n");
  }
  {
    @exec_Chart;
    wait(0.1);
  }*
end
endmodule
system
Chart()
endsystem
