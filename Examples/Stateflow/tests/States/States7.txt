%type: module

module Chart():
begin
  ChartEL := [];
  Chart_st := "Chart_A";
  log("enA" + "\n");
  Chart_A_st := "Chart_A_A1";
  s := "enA1";
  log("enA1" + "\n");
  (
    Chart_ret := 0;
    if Chart_st == "Chart_A" then
      Chart_A_done := 0;
      Chart_st == "Chart_A" -> (
        if Chart_A_st == "Chart_A_A1" then
          log("exA1" + "\n");
          Chart_A_st == "Chart_A_A1" -> Chart_A_st := ""
        else
          skip
        endif;
        s := "exA";
        log("exA" + "\n");
        Chart_st == "Chart_A" -> Chart_st := "";
        Chart_st == "" -> (
          Chart_st := "Chart_A";
          log("enA" + "\n");
          Chart_A_st := "Chart_A_A1";
          s := "enA1";
          log("enA1" + "\n")
        );
        Chart_A_done := 1
      );
      Chart_ret := Chart_A_done;
      Chart_ret == 0 -> (
        if Chart_A_st == "Chart_A_A1" then
          Chart_ret := 0
        else
          skip
        endif
      )
    else
      skip
    endif;
    wait(0.1)
  )**
end
endmodule
