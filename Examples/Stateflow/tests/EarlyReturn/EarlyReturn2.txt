%type: module

module Chart():
procedure exec_Chart begin
  Chart_ret := 0;
  if (Chart_st == "Chart_A") {
    Chart_A_done := 0;
    ChartEL != [] & top(ChartEL) == "E" --> {
      Chart_st == "Chart_A" --> {
        if (Chart_A_st == "Chart_A_A1") {
          log("ex_A1" + "\n");
          Chart_A_st == "Chart_A_A1" --> Chart_A_st := "";
        } else if (Chart_A_st == "Chart_A_A2") {
          log("ex_A2" + "\n");
          Chart_A_st == "Chart_A_A2" --> Chart_A_st := "";
        } else {
          skip;
        }
        s := "ex_A";
        log("ex_A" + "\n");
        Chart_st == "Chart_A" --> Chart_st := "";
        Chart_st == "" --> {
          Chart_st := "Chart_B";
          s := "en_B";
          log("en_B" + "\n");
        }
        Chart_A_done := 1;
      }
    }
    Chart_ret := Chart_A_done;
    Chart_ret == 0 --> {
      if (Chart_A_st == "Chart_A_A1") {
        Chart_A_A1_done := 0;
        s := "E";
        log("E" + "\n");
        ChartEL := push(ChartEL,"E");
        @exec_Chart;
        ChartEL := pop(ChartEL);
        Chart_A_st == "Chart_A_A1" --> {
          log("ex_A1" + "\n");
          Chart_A_st == "Chart_A_A1" --> Chart_A_st := "";
          s := "ta";
          log("ta" + "\n");
          Chart_A_st == "" & Chart_st == "Chart_A" --> {
            Chart_A_st := "Chart_A_A2";
            s := "en_A2";
            log("en_A2" + "\n");
          }
          Chart_A_A1_done := 1;
        }
        Chart_ret := Chart_A_A1_done;
      } else if (Chart_A_st == "Chart_A_A2") {
        Chart_ret := 0;
      } else {
        skip;
      }
    }
  } else if (Chart_st == "Chart_B") {
    Chart_ret := 0;
  } else {
    skip;
  }
end

begin
  ChartEL := [];
  Chart_st := "Chart_A";
  log("en_A" + "\n");
  Chart_A_st := "Chart_A_A1";
  s := "en_A1";
  log("en_A1" + "\n");
  {
    @exec_Chart;
    wait(0.1);
  }*
end
endmodule
system
Chart()
endsystem
