%type: module

module P0(p2c,c2p,init_x):
  output x;
  begin
    x := init_x;
    (<x_dot = 1 & true> |> [](p2c!x --> skip); c2p?x)**
  end
endmodule

module P1(p2c,c2p):
  begin
    (wait(2); p2c?x; c2p!x-1)**
  end
endmodule

system
  P0a = P0(ch1,ch2,$0) ||
  P1a = P1(ch1,ch2) ||
  P0b = P0(ch3,ch4,$1) ||
  P1b = P1(ch3,ch4)
endsystem
