%type: module

module Chart():
begin
  ChartEL := [];
  global_x_in?global_x;
  global_y_in?global_y;
  v := global_y;
  global_x := global_x + v;
  Chart_st := "Chart_A";
  global_x_out!global_x;
  global_y_out!global_y;
  (
    global_x_in?global_x;
    global_y_in?global_y;
    Chart_ret := 0;
    if Chart_st == "Chart_A" then
      Chart_A_done := 0;
      v := global_x + 1;
      global_x := v;
      log("%d %d\n",global_x,global_y);
      Chart_st == "Chart_A" -> (
        Chart_st := "Chart_B";
        Chart_A_done := 1
      );
      Chart_ret := Chart_A_done
    elif Chart_st == "Chart_B" then
      Chart_B_done := 0;
      v := global_y + 2;
      global_y := v;
      log("%d %d\n",global_x,global_y);
      Chart_st == "Chart_B" -> (
        Chart_st := "Chart_A";
        Chart_B_done := 1
      );
      Chart_ret := Chart_B_done
    else
      skip
    endif;
    global_x_out!global_x;
    global_y_out!global_y;
    wait(0.1)
  )**
end
endmodule
module data_ctrl():
begin
  global_x := 1;
  global_y := 2;
  (
    global_x_in!global_x -->
      skip $
    global_x_out?global_x -->
      skip $
    global_y_in!global_y -->
      skip $
    global_y_out?global_y -->
      skip
  )**
end
endmodule
system
Chart = Chart() ||
data_ctrl = data_ctrl()
endsystem
