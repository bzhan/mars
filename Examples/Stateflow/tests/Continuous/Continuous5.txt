begin
  ChartEL := [];

  Chart_A_st := "Chart_A_A1";
  s := "enA1";
  log("enA1" + "\n");
  (
    Chart_ret := 0;
    if Chart_st == "Chart_A" then
      Chart_A_done := 0;
      x >= 2 -> (
        Chart_st == "Chart_A" -> (
          if Chart_A_st == "Chart_A_A1" then
            Chart_A_st == "Chart_A_A1" -> Chart_A_st := ""
          elif Chart_A_st == "Chart_A_A2" then
            Chart_A_st == "Chart_A_A2" -> Chart_A_st := ""
          else
            skip
          endif;
          s := "exA";
          log("exA" + "\n");
          Chart_st == "Chart_A" -> Chart_st := "";
          Chart_st == "" -> (
            Chart_st := "Chart_B";
            s := "enB";
            log("enB" + "\n");
            x := 0
          );
          Chart_A_done := 1
        )
      );
      Chart_ret := Chart_A_done;
      Chart_ret == 0 -> (
        if Chart_A_st == "Chart_A_A1" then
          Chart_A_A1_done := 0;
          x >= 1 -> (
            Chart_A_st == "Chart_A_A1" -> (
              Chart_A_st := "";
              Chart_st == "Chart_A" -> (
                Chart_A_st := "Chart_A_A2";
                s := "enA2";
                log("enA2" + "\n")
              );
              Chart_A_A1_done := 1
            )
          );
          Chart_ret := Chart_A_A1_done
        elif Chart_A_st == "Chart_A_A2" then
          Chart_ret := 0
        else
          skip
        endif
      )
    elif Chart_st == "Chart_B" then
      Chart_B_done := 0;
      x >= 1 -> (
        Chart_st == "Chart_B" -> (
          Chart_st := "Chart_B";
          s := "enB";
          log("enB" + "\n");
          x := 0;
          Chart_B_done := 1
        )
      );
      Chart_ret := Chart_B_done
    else
      skip
    endif;
    if Chart_st == "Chart_A" then
      if Chart_A_st == "Chart_A_A1" then
        <x_dot = 1 &
          x < 2 &&
          x < 1
        >
      elif Chart_A_st == "Chart_A_A2" then
        <x_dot = 1 & x < 2>
      else
        skip
      endif
    elif Chart_st == "Chart_B" then
      <x_dot = 1 & x < 1>
    else
      skip
    endif
  )**
end
