%type: module

module Chart():
procedure exec_Chart begin
  Chart_ret := 0;
  if (Chart_A_st == "Chart_A_A1") {
    Chart_A_st := "";
    ChartEL := push(ChartEL,"E");
    @exec_Chart;
    ChartEL := pop(ChartEL);
    if (Chart_A_st == "") {
      Chart_A_st := "Chart_A_A2";
    }
  }
  Chart_ret := 0;
  if (Chart_B_st == "Chart_B_B1") {
    Chart_B_B1_done := 0;
    if ((ChartEL != [] && top(ChartEL) == "E") && x > 0) {
      x := x - 1;
      s := "a";
      log("a" + " %d\n",x);
      ChartEL := push(ChartEL,"E");
      @exec_Chart;
      ChartEL := pop(ChartEL);
      if (Chart_B_st == "Chart_B_B1") {
        Chart_B_st := "Chart_B_B2";
        s := "en_A2";
        log("en_A2" + " %d\n",x);
        Chart_B_B1_done := 1;
      }
    }
    Chart_ret := Chart_B_B1_done;
  } else if (Chart_B_st == "Chart_B_B2") {
    Chart_ret := 0;
  }
end

begin
  ChartEL := [];
  Chart_A_st := "Chart_A_A1";
  x := 6;
  Chart_B_st := "Chart_B_B1";
  {
    @exec_Chart;
    wait(0.1);
  }*
end
endmodule

system
Chart()
endsystem
